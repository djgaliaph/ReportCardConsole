public class ReportCard {
    
    // Method to display a message with the grade level
    public void displayMessage(String gradeLevel) {
        System.out.println("--- Report Card Information ---");
        System.out.println("Report card generated for grade level: " + gradeLevel);
        System.out.println("Good luck with your studies!");
    }
}

    public static void main(String[] args) {
